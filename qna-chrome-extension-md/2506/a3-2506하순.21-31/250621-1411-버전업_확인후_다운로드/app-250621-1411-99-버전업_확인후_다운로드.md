    <script>
        // Cursor 다운로드 페이지 URL
        const CURSOR_DOWNLOAD_URL = 'https://www.cursor.com/downloads';
        
        // Windows용 다운로드 링크 (실제로는 동적으로 가져와야 함)
        const WINDOWS_DOWNLOAD_URL = 'https://download.cursor.sh/windows';

        async function checkVersion() {
            const currentVersion = document.getElementById('currentVersion').value.trim();
            
            if (!currentVersion) {
                alert('현재 버전을 입력해주세요.');
                return;
            }

            showLoading(true);
            
            try {
                // 실제로는 Cursor API나 웹페이지에서 버전 정보를 가져와야 하지만,
                // CORS 정책으로 인해 직접 가져올 수 없으므로 시뮬레이션
                const latestVersion = await getLatestVersion();
                
                displayVersionInfo(currentVersion, latestVersion);
                checkForUpdates(currentVersion, latestVersion);
                
            } catch (error) {
                console.error('버전 확인 중 오류:', error);
                showError('버전 정보를 가져오는 중 오류가 발생했습니다.');
            } finally {
                showLoading(false);
            }
        }

        async function getLatestVersion() {
            // 실제 구현에서는 Cursor API를 호출하거나 웹페이지를 파싱해야 함
            // 현재는 시뮬레이션으로 1.1.3 반환
            await new Promise(resolve => setTimeout(resolve, 2000)); // 로딩 시뮬레이션
            
            // 실제로는 다음과 같은 방법을 사용할 수 있습니다:
            // 1. Cursor의 GitHub releases API 사용
            // 2. Cursor 웹사이트의 버전 정보 파싱
            // 3. Cursor의 공식 API 사용 (있다면)
            
            return '1.1.3';
        }

        function displayVersionInfo(currentVersion, latestVersion) {
            document.getElementById('currentVersionDisplay').textContent = currentVersion;
            document.getElementById('latestVersionDisplay').textContent = latestVersion;
            document.getElementById('versionInfo').style.display = 'block';
        }

        function checkForUpdates(currentVersion, latestVersion) {
            const statusDiv = document.getElementById('statusMessage');
            const downloadSection = document.getElementById('downloadSection');
            
            if (compareVersions(currentVersion, latestVersion) < 0) {
                // 업데이트 필요
                statusDiv.className = 'status update-available';
                statusDiv.innerHTML = `
                    <strong>업데이트 가능!</strong><br>
                    새 버전 ${latestVersion}이(가) 사용 가능합니다.
                `;
                downloadSection.style.display = 'block';
                
                // 다운로드 링크 설정
                const downloadBtn = document.getElementById('downloadBtn');
                downloadBtn.href = WINDOWS_DOWNLOAD_URL;
                downloadBtn.textContent = `Cursor ${latestVersion} 다운로드`;
                
            } else {
                // 최신 버전
                statusDiv.className = 'status up-to-date';
                statusDiv.innerHTML = `
                    <strong>최신 버전!</strong><br>
                    이미 최신 버전을 사용하고 있습니다.
                `;
                downloadSection.style.display = 'none';
            }
        }

        function compareVersions(v1, v2) {
            const parts1 = v1.split('.').map(Number);
            const parts2 = v2.split('.').map(Number);
            
            for (let i = 0; i < Math.max(parts1.length, parts2.length); i++) {
                const part1 = parts1[i] || 0;
                const part2 = parts2[i] || 0;
                
                if (part1 < part2) return -1;
                if (part1 > part2) return 1;
            }
            
            return 0;
        }

        function showLoading(show) {
            document.getElementById('loading').style.display = show ? 'block' : 'none';
        }

        function showError(message) {
            const statusDiv = document.getElementById('statusMessage');
            statusDiv.className = 'status error';
            statusDiv.innerHTML = `<strong>오류:</strong> ${message}`;
            document.getElementById('versionInfo').style.display = 'block';
        }

        function openDownloadPage() {
            window.open(CURSOR_DOWNLOAD_URL, '_blank');
        }

        // 페이지 로드 시 현재 버전 자동 감지 시도
        window.addEventListener('load', function() {
            // 실제로는 시스템에서 Cursor 버전을 감지하는 로직이 필요

            console.log('Cursor 버전 체커가 로드되었습니다.');
        });

        document.getElementById('currentVersion').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                checkVersion();
            }
        });

                return;

            try {
                // 여러 소스에서 최신 버전 정보를 가져오기 시도
                let latestVersion = await getLatestVersionFromGitHub();
                
                if (!latestVersion) {
                    addLog('GitHub API 실패, 대체 방법을 시도합니다...', 'info');
                    latestVersion = await getLatestVersionFromWebpage();
                }

                if (!latestVersion) {
                    // 모든 방법이 실패한 경우 시뮬레이션 데이터 사용
                    addLog('실제 버전 정보를 가져올 수 없어 시뮬레이션 데이터를 사용합니다.', 'error');
                    latestVersion = '1.1.3'; // 시뮬레이션
                }
                
                displayVersionInfo(currentVersion, latestVersion);
                checkForUpdates(currentVersion, latestVersion);
                
            } catch (error) {
                console.error('버전 확인 중 오류:', error);
                addLog(`버전 확인 중 오류: ${error.message}`, 'error');
                showError('버전 정보를 가져오는 중 오류가 발생했습니다.');
            } finally {
                showLoading(false);

            const parts1 = v1.split('.').map(Number);
            const parts2 = v2.split('.').map(Number);
            
            for (let i = 0; i < Math.max(parts1.length, parts2.length); i++) {
                const part1 = parts1[i] || 0;
                const part2 = parts2[i] || 0;
                
                if (part1 < part2) return -1;
                if (part1 > part2) return 1;
            }
            
            return 0;
        }

'.html' 이 있는 폴더에 'cursor-download' 라는 폴더를 만들고, 그 안에 마지막 설치파일을 보관해.
그리고, 버전을 확인할때는 무조건 하위폴더인 'cursor-download' 에서 현재 버전을 확인하는거지.
그리고, 최신버전을 다운로드 받게되면, 저 'cursor-download' 폴더에 여러 파일이 있게되니,

        // 자동 버전 감지
        async function autoDetectVersion() {
            addLog('현재 버전을 자동으로 감지하는 중...', 'info');
            
            if (currentFiles.length > 0) {
                // 가장 최신 파일의 버전을 사용
                const latestFile = currentFiles.reduce((latest, current) => {
                    return compareVersions(current.version, latest.version) > 0 ? current : latest;
                });
                
                document.getElementById('currentVersionDisplay').textContent = latestFile.version;
                addLog(`자동 감지된 버전: ${latestFile.version}`, 'success');
                
                // 자동으로 최신 버전 확인
                await checkVersion();
            } else {
                addLog('설치 파일을 찾을 수 없습니다.', 'error');
            }

        // 버전 확인
        async function checkVersion() {
            const currentVersion = document.getElementById('currentVersionDisplay').textContent;
            
            if (currentVersion === '확인 중...') {
                addLog('먼저 현재 버전을 감지해주세요.', 'error');
                return;
            }

            showLoading(true);
            addLog('GitHub에서 최신 버전을 확인하는 중...', 'info');

            try {
                latestVersion = await getLatestVersion();
                
                if (latestVersion) {
                    document.getElementById('latestVersionDisplay').textContent = latestVersion;
                    displayResult(currentVersion, latestVersion);
                    addLog(`최신 버전 확인됨: ${latestVersion}`, 'success');
                } else {
                    // API 실패 시 시뮬레이션 데이터 사용
                    latestVersion = '1.1.3';
                    document.getElementById('latestVersionDisplay').textContent = latestVersion;
                    displayResult(currentVersion, latestVersion);
                    addLog('GitHub API에 접근할 수 없어 시뮬레이션 데이터를 사용합니다.', 'error');
                }
            } catch (error) {
                console.error('오류:', error);
                addLog(`버전 확인 중 오류: ${error.message}`, 'error');
                showResult('버전 확인 중 오류가 발생했습니다.', 'error');
            } finally {
                showLoading(false);
            }
        }

        // GitHub API를 통한 최신 버전 확인
        async function getLatestVersion() {
            try {
                const response = await fetch('https://api.github.com/repos/getcursor/cursor/releases/latest');
                
                if (!response.ok) {
                    throw new Error(`API 오류: ${response.status}`);
                }
                
                const data = await response.json();
                return data.tag_name.replace('v', '');
            } catch (error) {
                console.error('GitHub API 호출 실패:', error);
                return null;
            }
        }

        // 결과 표시
        function displayResult(currentVersion, latestVersion) {
            document.getElementById('currentVersionResult').textContent = currentVersion;
            document.getElementById('latestVersionResult').textContent = latestVersion;
            document.getElementById('versionInfo').style.display = 'block';

            const needsUpdate = compareVersions(currentVersion, latestVersion) < 0;
            
            if (needsUpdate) {
                showResult(`업데이트 가능! 새 버전 ${latestVersion}이(가) 사용 가능합니다.`, 'update-available');
                setupDownloadLinks(latestVersion);
                document.getElementById('downloadLinks').style.display = 'block';
            } else {
                showResult('이미 최신 버전을 사용하고 있습니다.', 'up-to-date');
                document.getElementById('downloadLinks').style.display = 'none';
            }
        }

        // 자동 다운로드 및 정리
        async function downloadAndOrganize() {
            addLog('자동 다운로드 및 파일 정리를 시작합니다...', 'info');
            
            try {
                // 1. 이전 파일들을 old-files로 이동
                await moveOldFiles();
                
                // 2. 새 파일 다운로드 (시뮬레이션)
                addLog('새 버전을 다운로드하는 중...', 'info');
                await simulateDownload();
                
                // 3. 파일 목록 업데이트
                await scanCurrentFiles();
                await scanOldFiles();
                
                addLog('다운로드 및 정리가 완료되었습니다.', 'success');
                
            } catch (error) {
                addLog(`자동 다운로드 중 오류: ${error.message}`, 'error');
            }
        }

        // 이전 파일 이동
        async function moveOldFiles() {
            addLog('이전 파일들을 old-files 폴더로 이동하는 중...', 'info');
            
            if (currentFiles.length > 1) {
                // 가장 최신 파일을 제외한 나머지를 이동
                const filesToMove = currentFiles.slice(0, -1);
                
                filesToMove.forEach(file => {
                    addLog(`${file.name}을 old-files로 이동`, 'info');
                });
                
                addLog(`${filesToMove.length}개의 파일을 이동했습니다.`, 'success');
            } else {
                addLog('이동할 파일이 없습니다.', 'info');
            }
        }

        // 이전 파일 삭제
        async function cleanOldFiles() {
            addLog('old-files 폴더의 파일들을 삭제하는 중...', 'info');
            
            if (oldFiles.length > 0) {
                oldFiles.forEach(file => {
                    addLog(`${file.name} 삭제`, 'info');
                });
                
                oldFiles = [];
                displayOldFiles();
                addLog('이전 파일들을 삭제했습니다.', 'success');
            } else {
                addLog('삭제할 파일이 없습니다.', 'info');
            }
        }

        // 다운로드 시뮬레이션
        async function simulateDownload() {
            return new Promise((resolve) => {
                let progress = 0;
                const progressBar = document.createElement('div');
                progressBar.className = 'progress-bar';
                progressBar.innerHTML = '<div class="progress-fill"></div>';
                
                const progressFill = progressBar.querySelector('.progress-fill');
                
                // 진행률 표시를 위해 임시로 결과 섹션에 추가
                const resultDiv = document.getElementById('result');
                resultDiv.appendChild(progressBar);
                
                const interval = setInterval(() => {
                    progress += Math.random() * 20;
                    if (progress >= 100) {
                        progress = 100;
                        clearInterval(interval);
                        resultDiv.removeChild(progressBar);
                        addLog('다운로드 완료!', 'success');
                        resolve();
                    }
                    progressFill.style.width = progress + '%';
                }, 200);
            });
        }

files = glob.glob(os.path.join(DOWNLOAD_DIR, '*.exe'))

    print('설치파일이 없습니다.')

# 최신 파일(수정일 기준)
latest_file = max(files, key=os.path.getmtime)

# 최신 파일을 제외한 나머지 old-files로 이동
for f in files:
