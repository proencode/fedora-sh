        function getWeekDays(date) {
            const days = [];
            const sunday = new Date(date);
            sunday.setDate(date.getDate() - date.getDay());

            for (let i = 0; i < 7; i++) {
                const newDate = new Date(sunday);
                newDate.setDate(sunday.getDate() + i);
                days.push(newDate);

            return days;

        function formatDate(date) {
            const day = date.getDate();
            const month = date.getMonth() + 1;
            const weekdays = ['일', '월', '화', '수', '목', '금', '토'];
            const weekday = weekdays[date.getDay()];

            return `${month} / ${day} (${weekday})`;
        }

        function isToday(date) {
            const today = new Date();
            return date.getDate() === today.getDate() &&
                   date.getMonth() === today.getMonth() &&
                   date.getFullYear() === today.getFullYear();
        }

            return `${date.getFullYear()}-${date.getMonth() + 1}-${date.getDate()}`;
        }

        function createWeekCalendar(date) {
            const calendar = document.createElement('div');
            calendar.className = 'calendar';

            const days = getWeekDays(date);
            days.forEach(date => {
                const dayElement = document.createElement('div');
                dayElement.className = 'day';
                if (isToday(date)) {
                    dayElement.classList.add('today');
                }
                dayElement.textContent = formatDate(date);
                dayElement.onclick = () => openModal(date);
                calendar.appendChild(dayElement);
            });

            return calendar;

        function createMemoRow(date) {
            const memoRow = document.createElement('div');
            memoRow.className = 'memo-row';

            const days = getWeekDays(date);
            days.forEach(date => {
                const memoCell = document.createElement('div');
                memoCell.className = 'memo-cell';
                const dateKey = getDateKey(date);
                const memo = memos[dateKey] || '';

                if (isSimpleMemo && memo.length > 20) {
                    memoCell.textContent = memo.substring(0, 20) + '...';

                    memoCell.textContent = memo;

        function createCalendar(date) {
            const container = document.getElementById('calendarContainer');
            const yearDisplay = document.getElementById('yearDisplay');

            yearDisplay.textContent = `${date.getFullYear()} 년`;
            container.innerHTML = '';

            for (let i = -1; i <= 1; i++) {
                const weekDate = new Date(date);
                weekDate.setDate(date.getDate() + (i * 7));
                container.appendChild(createWeekCalendar(weekDate));
                container.appendChild(createMemoRow(weekDate));
            }

            currentDate.setDate(currentDate.getDate() + (weeks * 7));
            createCalendar(currentDate);

            const modal = document.getElementById('memoModal');
            modal.style.display = 'none';

                memos[getDateKey(selectedDate)] = memoText;
                saveMemosToStorage();
                createCalendar(currentDate);
                closeModal();

이 결과로 받은 소스에서 수정할 것은 다음 두가지야.

1. 날짜와 메모가 들어가는 박스의 테두리 두께가 한가지 두께로 통일이 돼야해.
그런데, 날짜가 들어있는 박스만 선이 두꺼워. 그러니 모든 선을 그냥 한가닥으로만 표현을 해줘.

2. 날짜를 표시하는 처음의 요청은 다음과 같아.

7개의 칸에 날짜를 표시하는데, 표시하는 내용은 날짜와 요일만 있으면 돼.
표시하는 방법은 "7 (토)" 와 같이 날짜와 괄호를 친 안에 요일을 표시하면 돼.
그런데, 나중에 보면 몇월인지 모를수도 있으니, 일요일에는 "6 / 7 (일)" 처럼 월도 표시해줘.
또한, 월이 바뀔때도 "7 / 1 (화)" 처럼 월표시를 해줘.
그리고 모든 글자표시는 한글로 해줘.
칸의 테두리는 한가지 두께로 적용해주고, 글자는 가로로든 세로로든 모두 중앙에 위치하도록 해줘.

그러니, 저 소스를 이 요청대로 수정을 해서 보여줘.

            const yearDisplay = document.getElementById('yearDisplay');
            yearDisplay.textContent = `${date.getFullYear()} 년`;
            
            const calendarGrid = document.createElement('div');
            calendarGrid.className = 'calendar-grid';
            
            for (let i = -1; i <= 1; i++) {
                const weekDate = new Date(date);
                weekDate.setDate(date.getDate() + (i * 7));
                const days = getWeekDays(weekDate);
                
                days.forEach(date => {
                    const dayElement = document.createElement('div');
                    dayElement.className = 'day';
                    if (isToday(date)) {
                        dayElement.classList.add('today');
                    }
                    dayElement.textContent = formatDate(date);
                    dayElement.onclick = () => openModal(date);
                    calendarGrid.appendChild(dayElement);
                });
                
                days.forEach(date => {
                    const memoCell = document.createElement('div');
                    memoCell.className = 'memo-cell';
                    const dateKey = getDateKey(date);
                    const memo = memos[dateKey] || '';
                    
                    if (isSimpleMemo && memo.length > 20) {
                        memoCell.textContent = memo.substring(0, 20) + '...';
                    } else {
                        memoCell.textContent = memo;
                    }
                    
                    calendarGrid.appendChild(memoCell);
                });
            }
            
            container.innerHTML = '';
            container.appendChild(calendarGrid);
        }

        function moveWeeks(weeks) {
            currentDate.setDate(currentDate.getDate() + (weeks * 7));
            createCalendar(currentDate);
        }

        function toggleSimpleMemo() {
            isSimpleMemo = !isSimpleMemo;
            const toggleButton = document.getElementById('simpleMemoToggle');
            toggleButton.classList.toggle('active');
            toggleButton.textContent = isSimpleMemo ? '단축메모' : '상세메모';
            createCalendar(currentDate);
        }

        function openModal(date) {
            selectedDate = date;
            const modal = document.getElementById('memoModal');
            const modalTitle = document.getElementById('modalTitle');
            const memoText = document.getElementById('memoText');

            modalTitle.textContent = formatDate(date);
            memoText.value = memos[getDateKey(date)] || '';
            modal.style.display = 'block';
        }

        function closeModal() {
            const modal = document.getElementById('memoModal');
            modal.style.display = 'none';
            selectedDate = null;
        }

        function saveMemo() {
            if (selectedDate) {
                const memoText = document.getElementById('memoText').value;
                memos[getDateKey(selectedDate)] = memoText;
                saveMemosToStorage();
                createCalendar(currentDate);
                closeModal();
            }
        }

        function saveToFile() {
            const dataStr = JSON.stringify(memos, null, 2);
            const dataUri = 'data:application/json;charset=utf-8,'+ encodeURIComponent(dataStr);
            
            const exportFileDefaultName = `calendar_memo_${new Date().toISOString().split('T')[0]}.json`;
            
            const linkElement = document.createElement('a');
            linkElement.setAttribute('href', dataUri);
            linkElement.setAttribute('download', exportFileDefaultName);
            linkElement.click();
        }

        function loadFromFile(event) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    try {
                        const loadedMemos = JSON.parse(e.target.result);
                        memos = loadedMemos;
                        saveMemosToStorage();
                        createCalendar(currentDate);
                        alert('메모를 불러왔습니다.');
                    } catch (error) {
                        alert('파일 형식이 올바르지 않습니다.');
                    }
                };
                reader.readAsText(file);
            }
        }

        document.addEventListener('DOMContentLoaded', function() {
            loadMemosFromStorage();
        });
